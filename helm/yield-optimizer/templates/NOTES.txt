1. Get the application URL by running these commands:
{{- if .Values.ingress.enabled }}
{{- range $host := .Values.ingress.hosts }}
  {{- range .paths }}
  http{{ if $.Values.ingress.tls }}s{{ end }}://{{ $host.host }}{{ .path }}
  {{- end }}
{{- end }}
{{- else if contains "NodePort" .Values.monitor.service.type }}
  export NODE_PORT=$(kubectl get --namespace {{ .Release.Namespace }} -o jsonpath="{.spec.ports[0].nodePort}" services {{ include "yield-optimizer.fullname" . }}-monitor)
  export NODE_IP=$(kubectl get nodes --namespace {{ .Release.Namespace }} -o jsonpath="{.items[0].status.addresses[0].address}")
  echo http://$NODE_IP:$NODE_PORT
{{- else if contains "LoadBalancer" .Values.monitor.service.type }}
     NOTE: It may take a few minutes for the LoadBalancer IP to be available.
           You can watch the status of by running 'kubectl get --namespace {{ .Release.Namespace }} svc -w {{ include "yield-optimizer.fullname" . }}-monitor'
  export SERVICE_IP=$(kubectl get svc --namespace {{ .Release.Namespace }} {{ include "yield-optimizer.fullname" . }}-monitor --template "{{"{{ range (index .status.loadBalancer.ingress 0) }}{{.}}{{ end }}"}}")
  echo http://$SERVICE_IP:{{ .Values.monitor.service.port }}
{{- else if contains "ClusterIP" .Values.monitor.service.type }}
  export POD_NAME=$(kubectl get pods --namespace {{ .Release.Namespace }} -l "app.kubernetes.io/name={{ include "yield-optimizer.name" . }},app.kubernetes.io/instance={{ .Release.Name }},app.kubernetes.io/component=monitor" -o jsonpath="{.items[0].metadata.name}")
  export CONTAINER_PORT=$(kubectl get pod --namespace {{ .Release.Namespace }} $POD_NAME -o jsonpath="{.spec.containers[0].ports[0].containerPort}")
  echo "Visit http://127.0.0.1:8080 to use your application"
  kubectl --namespace {{ .Release.Namespace }} port-forward $POD_NAME 8080:$CONTAINER_PORT
{{- end }}

2. Check the monitor service status:
   kubectl get pods --namespace {{ .Release.Namespace }} -l app.kubernetes.io/component=monitor

3. View monitor service logs:
   kubectl logs --namespace {{ .Release.Namespace }} -l app.kubernetes.io/component=monitor -f

4. Check metrics endpoint:
   kubectl --namespace {{ .Release.Namespace }} port-forward service/{{ include "yield-optimizer.fullname" . }}-monitor 9090:9090
   # Then visit http://localhost:9090/metrics
